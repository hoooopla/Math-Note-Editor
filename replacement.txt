    const handleClick = (e: React.MouseEvent) => {
        e.stopPropagation();
        if (e.metaKey || e.ctrlKey) {
            useStore.getState().openBlockInTab(targetBlock!.id, true);
        } else {
            useStore.getState().setActiveBlock(null);
            view?.contentDOM.blur();
            toggleOpen(e);
        }
    };
